#include <stdio.h>

int main()
{
	int s, t;
	int sum = 0;
	int iFalseCount = 0;
	int iTrueCount = 0;

	while (1)
	{
		printf("두 수를 입력하세요 : \n");

		printf("첫번째 : ");
		scanf("%d", &s);

		printf("두번째 : ");
		scanf("%d", &t);

		if (s > t)
		{
			printf("다시 입력하세요(첫번째 <= 두번째)");
			continue;
		}

		for (int i = s; i <= t; ++i)
		{
			if (i % 2 == 0 || i % 3 == 0)
			{
				sum += i;
			}

			if (i % 2 == 0 || i % 3 == 0)
			{
				++iTrueCount;
			}

			if (i % 2 != 0 && i % 3 != 0)
			{
				++iFalseCount;
			}
		}

		printf("합 : %d\n", sum);
		printf("2, 3배수인 것의 개수 : %d\n", iTrueCount);
		printf("2, 3배수가 아닌 것의 개수 : %d", iFalseCount);

		break;
	}

	return 0;
}
