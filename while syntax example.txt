#include <stdio.h>

enum M_STUDY
{
	M_NONE,
	M_1,
	M_2,
	M_3
};

void Menu()
{
	printf("============== Study ===========\n");
	printf("1. Question1\n");
	printf("2. Question2\n");
	printf("3. Question3\n");
	printf("메뉴를 선택하세요 : ");
}

int main()
{
	int iNum = 0;	

	while (1)
	{
		system("cls");
		Menu();
		scanf("%d", &iNum);

		if (iNum <= M_NONE || iNum > M_3)
		{
			printf("1~3사이의 수를 입력하세요\n");
			system("pause");
			continue;
		}

		switch (iNum)
		{
		case M_1:
			system("cls");

			int iNum1 = 0;
			int iSum1 = 0;
			int iCount1 = 0;
			
			printf("첫번째문제\n");

			while (iCount1 < 10)
			{				
				printf("정수를 입력하세요 : ");
				scanf("%d", &iNum1);
				printf("\n");

				if (iNum1 % 2 == 1)
				{
					iSum1 += iNum1;
					++iCount1;
				}
			}

			printf("홀수 10개의 합은 : %d 입니다. \n", iSum1);
			break;
		case M_2:
			system("cls");
			
			int iNum2 = 0;
			int iSum2 = 0;

			printf("두번째문제\n");

			while (1)
			{				
				printf("정수를 입력하세요 : ");
				scanf("%d", &iNum2);
				printf("\n");

				if (iNum2 == -1)
				{
					printf("총 합은 %d 입니다. \n", iSum2);
					break;
				}

				iSum2 += iNum2;
				
			}			
			break;
		case M_3:
			system("cls");
			
			int iNum3 = 0;
			int iSum3 = 0;
			int iCount3 = 0;

			printf("세번째문제\n");

			while (1)
			{
				printf("정수를 입력하세요 : ");
				scanf("%d", &iNum3);
				printf("\n");

				iSum3 += iNum3;
				++iCount3;

				if (iSum3 == 0)
					break;
			}

			printf("입력된 정수의 개수 : %d", iCount3);
			break;
		}
		
		break;
	}
	return 0;
}
