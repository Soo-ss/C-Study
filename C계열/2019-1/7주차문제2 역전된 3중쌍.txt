
#pragma warning (disable : 4996)

#include <stdio.h>

#define MAX 100

int main()
{
	int n, arr[MAX], count = 0;

	scanf("%d", &n);

	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &arr[i]);
	}

	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < n; ++j) // j = i + 1ÇØµµµÊ
		{
			for (int k = 0; k < n; ++k) // k = j + 1ÇØµµµÊ
			{
				if ((i < j && j < k) && (arr[i] > arr[j] && arr[j] > arr[k]))
					++count;
			}
		}
	}

	printf("%d\n", count);

	return 0;
}
